61  public int onStartCommand(Intent intent, int flags, int startId) {
62  // Extract the service instruction
63  final String action = intent.getAction();
64 
65  // Build the message parameters
66  Bundle extras = intent.getExtras();
67  if (extras == null)
68  extras = new Bundle();
69  extras.putString(TekCast.EXTRA_SERVICE_ACTION, action);
70 
71  // Run each command in the background thread.
72  final Message msg = mServiceHandler.obtainMessage();
73  msg.arg1 = startId;
74  msg.setData(extras);
75  mServiceHandler.sendMessage(msg);

131  @Override
132  public void handleMessage(Message msg) {
133  final Bundle data = msg.getData();
134  final ServiceAction action = ServiceAction.valueOf(data.getString(TekCast.EXTRA_SERVICE_ACTION));
135 
136  switch (action) {
137  case SEARCH:
138  // Search for Tekdaqcs
139  LocatorParams params = (LocatorParams) data.getSerializable(TekCast.EXTRA_LOCATOR_PARAMS);
140  if (params == null) {
141  params = LocatorParams.getDefaultInstance();
142  }
143  Locator.searchForTekDAQCS(mService, params);
144  break;
145  case STOP:
146  // Stop this service
147  mService.stopSelf();
148  break;
149  }
150  }
151  }

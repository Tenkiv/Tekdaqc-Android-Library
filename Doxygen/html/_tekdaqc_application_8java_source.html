72  /**
73  * Clears the known boards list and causes a new device locator packet to be sent. This
74  * effectively means that only un-connected boards will known to this list. Boards which
75  * were previously discovered and have an active connection will need to be tracked by
76  * the user application.
77  */
78  public final void refreshDeviceList() {
79  mBoards.clear();
80  final LocatorParams.Builder builder = new LocatorParams.Builder();
81  final Intent intent = new Intent(getApplicationContext(), DiscoveryService.class);
82  intent.setAction(DiscoveryService.ServiceAction.SEARCH.toString());
83  intent.putExtra(TekCast.EXTRA_LOCATOR_PARAMS, builder.build());
84  getApplicationContext().startService(intent);
85  }
